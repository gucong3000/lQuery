(function($){
	var q;
	
	function die(){
		if(q){
			q = q.die();
		}
		return false;
	}
	
	$(".run").each(function(){
		this.onclick = function(){
			die();
			q = $(".test2").each(function(){
				this.innerHTML = "test11";
			}).style("color: red;");
			return false;
		};
	});
	
	$(".die").each(function(){
		this.onclick = die;
	});

	$(".add").each(function(){
		this.onclick = function(){
			var doc = document,
				div = doc.createElement("div");
			div.setAttribute("class", "test2");
			div.className = "test2";
			div.innerHTML = "new node";
			doc.getElementById("testers").appendChild(div);
		};
